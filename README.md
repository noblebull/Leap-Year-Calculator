# Leap-Year-Calculator
#This code provides the basis for a simple leap year calculator in python
year = int(input("Enter an year"))
def leapyear(x)
  if x % 400 == 0:
    return True
  if x % 100 == 0:
    return False
  if x % 100 == 0:
    return True
  else:
    return False
print leapyear(year)
